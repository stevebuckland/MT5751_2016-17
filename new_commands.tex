% Question formatting
\newcounter{quno}
\renewcommand{\thequno}{\textbf{\arabic{quno}.}}
\newenvironment{questions}{%
  \begin{list}{\thequno}{\usecounter{quno}%
      \setlength{\parsep}{3mm}%
      \setlength{\itemsep}{26mm}%
      \setlength{\leftmargin}{15mm}%
      \setlength{\rightmargin}{0mm}%
      \setlength{\labelwidth}{3mm}%
      \setlength{\labelsep}{12mm}%
      \setlength{\listparindent}{5mm}}}{%
  \end{list}}

\newcounter{partno}
\renewcommand{\thepartno}{(\alph{partno})}
\newenvironment{parts}{%
  \begin{list}{\thepartno}{\usecounter{partno}
      \setlength{\parsep}{3mm}%
      \setlength{\itemsep}{3mm}%
      \setlength{\leftmargin}{8mm}%
      \setlength{\rightmargin}{0mm}%
      \setlength{\labelwidth}{5mm}%
      \setlength{\labelsep}{6mm}%
      \setlength{\listparindent}{0mm}}}{%
  \end{list}}
